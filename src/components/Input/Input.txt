2. 사용 방법 (중요)
    Login.js에서 사용
        import Input from "../../components/Input/Input";

        <Input
        label="이메일"
        type="text"
        value={id}
        onChange={(e) => setId(e.target.value)}
        placeholder="이메일 입력"
        />

        <Input
        label="비밀번호"
        type="password"
        value={pw}
        onChange={(e) => setPw(e.target.value)}
        placeholder="비밀번호 입력"
        />
